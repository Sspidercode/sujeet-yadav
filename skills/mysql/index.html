MySQL

1. Database Creation and Management

Create/drop databases; select for use. Use UTF-8 charset for international support. Manage users/privileges with GRANT/REVOKE.

-- Create database
        CREATE DATABASE sujeet_portfolio CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
        
        -- Use database
        USE sujeet_portfolio;
        
        -- Drop database
        DROP DATABASE IF EXISTS old_db;
        
        -- Create user and grant privileges
        CREATE USER 'sujeet'@'localhost' IDENTIFIED BY 'secure_password';
        GRANT ALL PRIVILEGES ON sujeet_portfolio.* TO 'sujeet'@'localhost';
        FLUSH PRIVILEGES;
2. Tables and Data Types

Tables store data with columns of types: INT, VARCHAR, TEXT, DATE, BOOLEAN. Use AUTO_INCREMENT for IDs; ENUM/SET for constrained values.

-- Create table
        CREATE TABLE users (
            id INT AUTO_INCREMENT PRIMARY KEY,
            name VARCHAR(100) NOT NULL,
            email VARCHAR(255) UNIQUE NOT NULL,
            age INT CHECK (age >= 18),
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            is_active BOOLEAN DEFAULT TRUE,
            role ENUM('admin', 'user', 'guest') DEFAULT 'user'
        );
        
        -- Insert sample data
        INSERT INTO users (name, email, age, role) VALUES 
        ('Sujeet Yadav', 'sujeet@example.com', 30, 'admin');
3. CRUD Operations

Create (INSERT), Read (SELECT), Update (UPDATE), Delete (DELETE). Use WHERE for specificity; LIMIT for pagination.

-- Create
        INSERT INTO users (name, email, age) VALUES ('Alice', 'alice@example.com', 25);
        
        -- Read
        SELECT id, name, email FROM users WHERE is_active = TRUE ORDER BY age DESC LIMIT 10;
        
        -- Update
        UPDATE users SET age = 31, is_active = FALSE WHERE id = 1;
        
        -- Delete
        DELETE FROM users WHERE id = 2;
4. Advanced Queries and JOINs

SELECT with functions (COUNT, AVG); JOIN for relationships (INNER, LEFT, RIGHT). Subqueries for complex filtering.

-- Create related table
        CREATE TABLE posts (
            id INT AUTO_INCREMENT PRIMARY KEY,
            user_id INT,
            title VARCHAR(200),
            FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
        );
        
        -- JOIN query
        SELECT u.name, p.title, COUNT(p.id) as post_count
        FROM users u
        LEFT JOIN posts p ON u.id = p.user_id
        GROUP BY u.id, u.name
        HAVING post_count > 0
        ORDER BY post_count DESC;
        
        -- Subquery
        SELECT name FROM users WHERE id IN (SELECT user_id FROM posts WHERE title LIKE '%Laravel%');
5. Indexing

Indexes speed up queries (B-tree default); PRIMARY/UNIQUE auto-indexed. Composite for multi-column; FULLTEXT for search. Monitor with SHOW INDEX.

-- Add indexes
        ALTER TABLE users ADD INDEX idx_email (email);
        ALTER TABLE posts ADD INDEX idx_user_title (user_id, title(50));
        
        -- Fulltext
        ALTER TABLE posts ADD FULLTEXT(title);
        
        -- Query with index hint
        SELECT * FROM users FORCE INDEX(idx_email) WHERE email = 'sujeet@example.com';
        
        -- View indexes
        SHOW INDEX FROM users;
6. Constraints and Relationships

Constraints: PRIMARY KEY, FOREIGN KEY, UNIQUE, CHECK. Enforce integrity; CASCADE for deletes/updates in relationships (1:1, 1:M, M:N).

-- Table with constraints
        CREATE TABLE orders (
            id INT PRIMARY KEY AUTO_INCREMENT,
            user_id INT NOT NULL,
            total DECIMAL(10,2) CHECK (total >= 0),
            FOREIGN KEY (user_id) REFERENCES users(id) ON UPDATE CASCADE ON DELETE RESTRICT,
            UNIQUE KEY unique_user_order (user_id, created_at)
        );
        
        -- Insert with constraint violation (will fail if email exists)
        INSERT INTO users (name, email) VALUES ('Duplicate', 'sujeet@example.com');
7. Transactions

Transactions ensure ACID: START TRANSACTION, COMMIT, ROLLBACK. Isolation levels (READ COMMITTED default) prevent anomalies.

START TRANSACTION;
        
        INSERT INTO users (name, email) VALUES ('Bob', 'bob@example.com');
        UPDATE posts SET title = 'Updated' WHERE user_id = LAST_INSERT_ID();
        
        -- If error, rollback
        -- ROLLBACK;
        
        COMMIT;  -- Success
        
        -- Isolation
        SET SESSION TRANSACTION ISOLATION LEVEL READ COMMITTED;
8. Query Optimization

Use EXPLAIN to analyze plans; optimize with indexes, LIMIT, avoid SELECT *. Slow query log for monitoring; InnoDB for row-level locking.

-- Explain query
        EXPLAIN SELECT * FROM users u JOIN posts p ON u.id = p.user_id WHERE u.age > 25;
        
        -- Optimized: Specific columns, index
        EXPLAIN SELECT u.name, p.title FROM users u USE INDEX(idx_age) JOIN posts p ON u.id = p.user_id WHERE u.age > 25 LIMIT 100;
        
        -- Show status
        SHOW STATUS LIKE 'Slow_queries';
        SHOW VARIABLES LIKE 'slow_query_log';
9. Stored Procedures and Functions

Procedures for reusable logic; functions for computations. Use DELIMITER for multi-statements; parameters IN/OUT.

DELIMITER //
        
        CREATE PROCEDURE GetUserPosts(IN userId INT)
        BEGIN
            SELECT u.name, p.title FROM users u JOIN posts p ON u.id = p.user_id WHERE u.id = userId;
        END //
        
        CREATE FUNCTION GetUserAge(birth_year INT) RETURNS INT
        DETERMINISTIC
        BEGIN
            RETURN YEAR(CURDATE()) - birth_year;
        END //
        
        DELIMITER ;
        
        -- Call
        CALL GetUserPosts(1);
        SELECT GetUserAge(1995);
10. Migrations and Backups

Mysqldump for backups; ALTER TABLE for schema changes. Version control with tools like Flyway or Laravel migrations.

-- Backup
        mysqldump -u sujeet -p sujeet_portfolio > backup.sql
        
        -- Restore
        mysql -u sujeet -p sujeet_portfolio < backup.sql
        
        -- Migration: Add column
        ALTER TABLE users ADD COLUMN bio TEXT AFTER email;
        
        -- Drop column
        ALTER TABLE users DROP COLUMN bio;
        
        -- Laravel migration example (up/down)
        public function up() {
            Schema::table('users', function (Blueprint $table) {
                $table->text('bio')->after('email');
            });
        }
        public function down() {
            Schema::table('users', function (Blueprint $table) {
                $table->dropColumn('bio');
            });
        }
