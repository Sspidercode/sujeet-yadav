JavaScript

1. Variables and Data Types

JavaScript uses let, const, and var to declare variables. Data types include primitives (string, number, boolean, null, undefined, symbol, bigint) and objects (arrays, functions, etc.). let and const are block-scoped (ES6+), while var is function-scoped. 

// Variables
let name = 'Sujeet';  // String (mutable)
const age = 30;       // Number (immutable)
var isDeveloper = true; // Boolean

// Data types check
console.log(typeof name);     // 'string'
console.log(typeof age);      // 'number'
console.log(typeof null);     // 'object' (historical bug)

// Example usage
if (age > 25) {
console.log(`${name} is experienced.`);
}
2. Operators

Operators include arithmetic (+, -, *, /, %), comparison (==, ===, >, <), logical (&&, ||, !), assignment (=, +=), and more. Use===for strict equality to avoid type coercion. 

let a = 10, b = 5;

// Arithmetic
console.log(a + b);  // 15
console.log(a % b);  // 0

// Comparison
console.log(a === '10');  // false (strict)
console.log(a == '10');   // true (loose)

// Logical
let isAdult = age >= 18 && isDeveloper;
console.log(isAdult);     // true
3. Control Structures

Control flow with if/else, switch, for, while, do-while loops. These manage execution based on conditions or repetition. 

// If/else
if (age > 18) {
console.log('Adult');
} else {
console.log('Minor');
}

// For loop
for (let i = 0; i < 3; i++) {
console.log(`Loop: ${i}`);
}

// While loop
let count = 0;
while (count < 2) {
console.log(count);
count++;
}

// Switch
switch (true) {
case age > 30:
  console.log('Senior');
  break;
default:
  console.log('Junior');
}
4. Functions

Functions are first-class objects. Declare with function, arrow functions (ES6), or expressions. Support parameters, defaults, rest params, and return values. 

// Function declaration
function greet(name) {
return `Hello, ${name}!`;
}

// Arrow function (ES6)
const add = (x, y) => x + y;

// With default params and rest
function sum(...numbers) {
return numbers.reduce((acc, num) => acc + num, 0);
}

console.log(greet('Sujeet'));  // 'Hello, Sujeet!'
console.log(add(2, 3));        // 5
console.log(sum(1, 2, 3));     // 6
5. Arrays

Arrays are ordered lists. Use methods like push, pop, map, filter, reduce for manipulation. ES6+ includes destructuring and spread. 

let skills = ['JS', 'HTML', 'CSS'];

// Basic operations
skills.push('React');  // Add
console.log(skills.pop());  // 'React' (remove last)

// ES6: Map (transform)
let upperSkills = skills.map(skill => skill.toUpperCase());
console.log(upperSkills);  // ['JS', 'HTML', 'CSS']

// Filter
let jsSkills = skills.filter(skill => skill.includes('J'));
console.log(jsSkills);  // ['JS']

// Reduce
let totalLength = skills.reduce((acc, skill) => acc + skill.length, 0);
console.log(totalLength);  // 9

// Destructuring
let [first, ...rest] = skills;
console.log(first);  // 'JS'
6. Objects

Objects store key-value pairs. Access via dot or bracket notation. ES6+ features: destructuring, spread, shorthand properties. 

let person = {
name: 'Sujeet',
age: 30,
skills: ['JS', 'Python']
};

// Access
console.log(person.name);     // 'Sujeet'
console.log(person['age']);   // 30

// Methods
person.greet = function() { return `Hi, I'm ${this.name}`; };
console.log(person.greet());  // 'Hi, I'm Sujeet'

// ES6: Destructuring
let { name, age } = person;
console.log(name, age);  // 'Sujeet' 30

// Spread
let updated = { ...person, city: 'Delhi' };
console.log(updated.city);  // 'Delhi'
7. DOM Manipulation

The DOM (Document Object Model) represents HTML as a tree. Use document methods to select, modify, create elements. 

// Select elements
let title = document.querySelector('.title-v2');
let button = document.createElement('button');
button.textContent = 'Click Me';
document.body.appendChild(button);

// Modify
title.textContent = 'Updated Title';
title.style.color = 'blue';

// Example: Update year in footer (assuming script.js does this)
document.getElementById('year').textContent = new Date().getFullYear();
8. Events

Events handle user interactions (click, keydown). Use addEventListener for attachment. Supports event bubbling/preventDefault. 

// Add event listener
button.addEventListener('click', function(event) {
console.log('Button clicked!');
event.preventDefault();  // Stop default if needed
});

// Example: Form submit
let form = document.querySelector('form');
form.addEventListener('submit', (e) => {
e.preventDefault();
alert('Form submitted!');
});
9. Async JavaScript

Handle asynchronous code with callbacks, Promises, async/await. Useful for APIs, timers. Avoid callback hell with Promises. 

// setTimeout (callback)
setTimeout(() => console.log('Delayed!'), 1000);

// Promise
let fetchData = () => new Promise((resolve, reject) => {
setTimeout(() => resolve('Data loaded'), 1000);
});
fetchData().then(data => console.log(data));

// Async/await (ES8)
async function getData() {
try {
  let data = await fetchData();
  console.log(data);
} catch (error) {
  console.error(error);
}
}
getData();
10. ES6+ Features

Modern JS includes modules, classes, template literals, let/const, destructuring, arrow functions, and more for cleaner code. 

// Template literals
let msg = `Age: ${age}`;  // `Age: 30`

// Classes
class Person {
constructor(name) {
  this.name = name;
}
greet() {
  return `Hello, ${this.name}`;
}
}
let sujeet = new Person('Sujeet');
console.log(sujeet.greet());

// Modules (export/import - in separate files)
export const PI = 3.14;
import { PI } from './math.js';
